def build_flow_svg(
    svg_template_path: str,
    flux_matrix: np.ndarray,
    mapping_svg_fluxes: dict[str, list[tuple[int, int]]],
    scale_max_px: float = 25.0,
) -> str:
    """
    Charge le template SVG et met √† jour les `stroke-width`
    en normalisant l‚Äô√©paisseur maximale √† `scale_max_px` pixels.
    Retourne le SVG en **cha√Æne de caract√®res**.
    """
    parser = etree.XMLParser(remove_blank_text=False)
    # try:
    #     tree = etree.parse(svg_template_path, parser)
    # except Exception as e:
    #     print("‚ùå Erreur durant le parsing du SVG :")
    #     print(f"   Chemin utilis√©  : {svg_template_path}")
    #     print(f"   Exception lev√©e : {repr(e)}")
    #     raise
    tree = etree.parse(svg_template_path, parser)
    root = tree.getroot()
    # # Force le SVG √† √™tre redimensionnable selon son viewBox
    # if "width" in root.attrib:
    #     del root.attrib["width"]
    # if "height" in root.attrib:
    #     del root.attrib["height"]
    # # On s‚Äôassure que le viewBox sera pris en compte √† 100 %
    # root.attrib["preserveAspectRatio"] = "xMinYMin meet"
    ns = {"svg": root.nsmap.get(None)}

    # Valeur max pour la normalisation
    vmax = max(
        (flux_matrix[i, j] for pairs in mapping_svg_fluxes.values() for i, j in pairs if flux_matrix[i, j] > 0),
        default=1e-9,
    )

    for path_id, ij_list in mapping_svg_fluxes.items():
        total = sum(float(flux_matrix[i, j]) for i, j in ij_list)
        width = 0 if total == 0 else total / vmax * scale_max_px

        nodes = root.xpath(f"//svg:path[@id='{path_id}']", namespaces=ns)
        if not nodes:

def build_flow_svg_full(
    svg_template_path: str,
    flux_matrix: np.ndarray,
    mapping_svg_fluxes: dict[str, list[tuple[int, int]]],
    scale_max_px: float = 25.0,
) -> str:
    """
    1) Met √† jour les `stroke-width` des chemins SVG selon flux_matrix (normalis√©).
    2) Calcule la bounding‚Äêbox de toutes les formes (paths + rects).
    3) R√©√©crit le <svg> pour supprimer width/height, ajouter viewBox et overflow="visible".
    4) Retourne le SVG complet (cha√Æne UTF‚Äê8), pr√™t √† √™tre inject√© tel quel.
    """

    # --- 1) parser le SVG ---
    parser = etree.XMLParser(remove_blank_text=False)
    tree = etree.parse(svg_template_path, parser)
    root = tree.getroot()
    ns = {"svg": root.nsmap.get(None)}

    # --- 2) Traiter les stroke‚Äêwidth comme avant ---
    # 2.1 Calculer vmax pour la normalisation
    all_flux = [flux_matrix[i, j] for pairs in mapping_svg_fluxes.values() for i, j in pairs if flux_matrix[i, j] > 0]
    vmax = max(all_flux) if all_flux else 1e-9

    # 2.2 Mettre √† jour chaque <path> cibl√©
    for path_id, ij_list in mapping_svg_fluxes.items():
        total = sum(float(flux_matrix[i, j]) for i, j in ij_list)
        width = 0 if total == 0 else (total / vmax) * scale_max_px

        nodes = root.xpath(f"//svg:path[@id='{path_id}']", namespaces=ns)
        if not nodes:
            continue
        path = nodes[0]
        style = path.attrib.get("style", "")
        if "stroke-width" in style:
            style = re.sub(r"stroke-width:[^;]+", f"stroke-width:{width}", style)
        else:
            style = style.rstrip(";") + f";stroke-width:{width}"
        path.attrib["style"] = style

    # --- 3) Calculer la bounding‚Äêbox de tout le contenu ---
    # On cherche minx, miny, maxx, maxy parmi:
    #   ‚Ä¢ Tous les ‚ÄòM x,y‚Äô, ‚ÄòL x,y‚Äô, ‚ÄòC ‚Ä¶‚Äô d‚Äôun <path> (on prend TOUTES les coordonn√©es absolues)
    #   ‚Ä¢ Tous les coins des <rect> (x,y) + (x+width, y+height)
    minx = miny = float("inf")
    maxx = maxy = float("-inf")

    # 3.1 Pour chaque <path> : extraire tous les nombres de ‚Äòd=‚Äô
    for elem in root.xpath("//svg:path", namespaces=ns):
        d = elem.attrib.get("d", "")
        # trouver toutes les valeurs num√©riques (x1, y1, x2, y2, ‚Ä¶)
        nums = re.findall(r"[-+]?\d*\.?\d+|[-+]?\d+", d)
        # on convertit en float et on prend paires (0,1), (2,3), ‚Ä¶
        coords = [float(x) for x in nums]
        xs = coords[0::2]
        ys = coords[1::2]
        if xs and ys:
            minx = min(minx, min(xs))
            maxx = max(maxx, max(xs))
            miny = min(miny, min(ys))
            maxy = max(maxy, max(ys))

    # 3.2 Pour chaque <rect> : coins (x, y) et (x+width, y+height)
    for elem in root.xpath("//svg:rect", namespaces=ns):
        x = float(elem.attrib.get("x", 0))
        y = float(elem.attrib.get("y", 0))
        w = float(elem.attrib.get("width", 0))
        h = float(elem.attrib.get("height", 0))
        minx = min(minx, x)
        maxx = max(maxx, x + w)
        miny = min(miny, y)
        maxy = max(maxy, y + h)

    # Si on n‚Äôa rien trouv√© (aucun path ni rect), on reste sur le viewBox existant (ou on quitte)
    if minx == float("inf"):
        # Rien √† calculer, on renvoie le SVG original modifi√©
        return etree.tostring(tree, encoding="utf-8", pretty_print=False).decode()

    # --- 4) R√©√©crire le <svg> pour englober ces bornes ---
    # 4.1 Supprimer width/height (pour que le viewBox gouverne l‚Äôaffichage)
    for attr in ("width", "height"):
        if attr in root.attrib:
            del root.attrib[attr]

    # 4.2 Ajouter le viewBox = "minx miny width_box height_box"
    vb_x = minx
    vb_y = miny
    vb_width = maxx - minx
    vb_height = maxy - miny
    root.attrib["viewBox"] = f"{vb_x} {vb_y} {vb_width} {vb_height}"

    # 4.3 S‚Äôassurer que rien n‚Äôest masqu√© : overflow="visible"
    root.attrib["overflow"] = "visible"

    tree.write(Path(svg_template_path).parent / "test.svg", pretty_print=False, xml_declaration=True, encoding="UTF-8")

    # --- 5) Renvoyer le SVG final (sans <?xml ‚Ä¶>) ---
    return etree.tostring(tree, encoding="utf-8", xml_declaration=False, pretty_print=False).decode()
